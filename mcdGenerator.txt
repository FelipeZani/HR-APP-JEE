Table Employee{
employeeId int [pk]
name String
lastname String
departementId int [ref : < Departement.departmentId]
postId int [ref: < Post.postId]
rank  String
}

Table Project{
projectId int [pk]
label String
status String
}

Table RoleEmployee{
roleId int [pk]
label String
permissions String
}

Table Departement{
departmentId int [pk]
name String
}

Table PayStub{
payStubId int [pk ]
total money
creationDate date
date date
employeeId int [ref: < Employee.employeeId]
}

Table Employee_Project{
employeeId int [ref: < Employee.employeeId]
projectId int [ref: < Project.projectId]
assignementDate date
}

Table Employee_RoleEmployee{
employeeId int [ref: < Employee.employeeId]
id_role int [ref: < RoleEmployee.roleId]
}

Table Post{
postId int [pk]
label String
wage money
}

Table Account {
  username String
  password  String
  firstConnection boolean
  employeeId int [ref : - Employee.employeeId]
}
